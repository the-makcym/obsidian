$$
\cosh x =
\frac{\sinh \pi}{\pi} \;
\left(
  1 +
  2\sum\limits_{n=1}^{\infty}
    \left[
      (-1)^n \,
      \frac{1}{1+n^2} \,
      \cos nx
    \right]
\right)
$$

$$
\sinh x =
\frac{2 \sinh \pi}{\pi} \;
\sum\limits_{n=1}^{\infty}
\left[
  (-1)^{n-1} \,
  \frac{n}{1+n^2} \,
  \sin nx
\right]
$$
